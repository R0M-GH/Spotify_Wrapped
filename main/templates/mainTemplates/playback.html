// Spotify Web Playback SDK Initialization let player; let deviceId; const token = '{{ access_token }}'; // Pass user's access token dynamically const topTracks = {{ top_tracks|safe }}; // Pass the list of tracks as a JSON-safe array let currentTrackIndex = 0; // Index of the currently playing track window.onSpotifyWebPlaybackSDKReady = () => { player = new Spotify.Player({ name: 'Web Playback SDK Player', getOAuthToken: cb => { cb(token); }, volume: 0.5 }); // Connect the player player.connect(); // Get the device ID player.addListener('ready', ({ device_id }) => { console.log('Ready with Device ID', device_id); deviceId = device_id; // Save the device ID for playback // Play the first track automatically playTrack(topTracks[currentTrackIndex].uri); }); // Log player state player.addListener('player_state_changed', state => { if (!state) return; // Detect when the track ends if (state.paused && state.position === 0) { playNextTrack(); } }); // Handle player errors player.addListener('initialization_error', ({ message }) => console.error(message)); player.addListener('authentication_error', ({ message }) => console.error(message)); player.addListener('account_error', ({ message }) => console.error(message)); player.addListener('playback_error', ({ message }) => console.error(message)); }; // Function to play a track function playTrack(trackUri) { player.activateElement().then(() => { // Activate playback fetch(`https://api.spotify.com/v1/me/player/play?device_id=${deviceId}`, { method: 'PUT', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ uris: [trackUri] // Play the selected track }) }).then(response => { if (response.status === 204) { console.log(`Now playing: ${trackUri}`); } else { console.error('Error playing track:', response); } }); }).catch(err => console.error('Error activating playback:', err)); } // Function to play the next track function playNextTrack() { currentTrackIndex++; // Increment the track index if (currentTrackIndex < topTracks.length) { playTrack(topTracks[currentTrackIndex].uri); // Play the next track } else { console.log('End of playlist. No more tracks to play.'); } }
